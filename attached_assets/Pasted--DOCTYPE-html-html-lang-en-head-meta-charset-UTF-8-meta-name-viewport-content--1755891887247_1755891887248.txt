  <script>
    // Optional: simple client-side search + sorting
    const q = document.getElementById('q');
    const sort = document.getElementById('sort');
    const grid = document.getElementById('grid');

    function cards(){ return [...grid.querySelectorAll('.brand-card')]; }

    q?.addEventListener('input', ()=>{
      const term = q.value.toLowerCase();
      cards().forEach(c=>{
        const text = c.querySelector('.brand-name').textContent.toLowerCase();
        c.style.display = text.includes(term) ? '' : 'none';
      });
    });

    sort?.addEventListener('change', ()=>{
      const list = cards().sort((a,b)=>{
        const A = a.querySelector('.brand-name').textContent.trim();
        const B = b.querySelector('.brand-name').textContent.trim();
        return sort.value === 'az' ? A.localeCompare(B) : 0; // placeholder for other options
      });
      list.forEach(el=>grid.appendChild(el));
    });
  </script>
